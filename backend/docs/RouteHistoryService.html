new RouteHistoryService()

Description:

  • Route History Service Manages storage and retrieval of completed route history for users Tracks route statistics and provides route management functionality

Source:
  • services/routeHistory.js, line 3
Example
// Save a completed route
await RouteHistoryService.saveRouteHistory({
  userId: 'user-123',
  from: 'Home',
  to: 'Clinic',
  mode: 'TRANSIT',
  duration: '30 mins',
  steps: 5
});
